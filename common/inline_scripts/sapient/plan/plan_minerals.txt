# --- Basic buffer treadmill (capped) --------------------------------------
# Climbs one district at a time up to the tech-tiered ceiling (sapient_minerals_buffer_over_cap), so the empire
# carries a real surplus cushion instead of budgeting to a knife's edge. Stops at
# the cap -- late game the buffer is noise, so it stops costing districts.
subplan = {
	scaling = yes
	set_name = "Scaling Minerals (Buffer)"
	potential = {
		is_nomadic = no
		# Tech-tiered stop: 100/150/250 income ceiling by mining tech rung
		# (t2 = the old flat 250, so late-game behaviour is unchanged).
		NOT = { sapient_minerals_buffer_over_cap = yes }
	}
	income = { minerals = @sapient_minerals_step }
}
