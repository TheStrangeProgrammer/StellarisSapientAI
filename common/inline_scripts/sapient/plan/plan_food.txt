# SapientAI — food demand: three distinct consumers, one file
#   1. eating (vanilla, uses_food)
#   2. industry input, catalytic archetype: base floor here, scaling inside the
#      alloy/CG chains (plan_alloys/plan_cg)
#   3. SHIPBUILDING, bioship empires: food is their alloys. Vanilla caps this at
#      500 while a normal late empire wants 1000+ alloys — a bioship fleet needs
#      food income at alloy scale (~4k late-game). Basic tier raises the ceiling
#      to @sapient_bioship_food_limit; the ladder tiers must continue the climb.

subplan = {
	set_name = "Base Food"
	potential = {
		country_uses_bio_ships = no
		country_uses_food = yes
	}
	
	income = {
		food = @economic_plan_food_target
	}
}

subplan = {
	scaling = yes
	set_name = "Base Food Scaling"
	potential = {
		country_uses_bio_ships = no
		country_uses_food = yes
		OR = {
			NOR = {
				has_monthly_income = { resource = food value >= @economic_plan_scaling_food_limit }
				resource_stockpile_compare = {
					resource = food

			}
		}
	}

		food = 20
	}
}
