# SapientAI — consumer goods chains
# Vanilla CG scaling was optional at +1/step to a 100 ceiling — the most anemic
# demand in the plan system, upstream of research. Replaced with coupled chains
# (artisan ratios: minerals 6->6 = 1.0, catalytic food 9->7.5 = 1.2; x1.25 buffer).
# Non-optional: CG starvation stalls researchers and unity jobs.

subplan = {
	set_name = "Base Consumer Goods"
	potential = {
		country_uses_consumer_goods = yes
	}
	income = {
		consumer_goods = @economic_plan_consumer_goods_target
	}
}

inline_script = {
	script = sapient/production_chain
	NAME = "Sapient CG Chain (Minerals)"
	GATE = sapient_cg_chain_minerals
	OUTPUT = consumer_goods
	INPUT = minerals
	OUT_STEP = 5
	IN_STEP = 7
	OUT_LIMIT = @sapient_cg_ceiling
}

inline_script = {
	script = sapient/production_chain
	NAME = "Sapient CG Chain (Food)"
	GATE = sapient_cg_chain_food
	OUTPUT = consumer_goods
	INPUT = food
	OUT_STEP = 5
	IN_STEP = 8
	OUT_LIMIT = @sapient_cg_ceiling
}

subplan = {
	scaling = yes
	set_name = "Base Obsessional CG Scaling"
	potential = {
		has_valid_civic = civic_machine_obsessional_directive
		has_monthly_income = { resource = consumer_goods value < @economic_plan_scaling_obsessional_consumer_goods_limit_base }
	}
	income = {
		consumer_goods = 20
	}
}
