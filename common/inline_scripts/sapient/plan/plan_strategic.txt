# Strategic resources — two branches per refinable, magnitude by PATH COST.
#   Harvesting (owns a deposit): efficient -> demand a modest target (builds zones).
#   Refinery only (has tech, no deposit): inefficient stopgap -> tiny trickle
#     just for critical upkeep, so the AI doesn't waste minerals converting.
# Each branch is per-resource and self-limits via its own income-reached clause.
# Exotics (zro/dark matter/nanites) have no lane: unbuildable to order.
# Inert deficit weights removed throughout.

# --- Volatile Motes -------------------------------------------------------
subplan = {
	set_name = "Strategic Volatile Motes (Harvesting)"
	potential = {
		# The gathering zone (shared_volatile_motes_zone) hard-requires this tech —
		# without it the deposit can't be worked, so demanding the SR here would sit
		# permanently unfulfilled and freeze scaling. Deposit alone is NOT enough.
		has_technology = tech_mine_volatile_motes
		any_owned_planet = { has_volatile_motes_deposit = yes }
		NOT = { has_monthly_income = { resource = volatile_motes value >= @sapient_sr_base } }
	}
	income = { volatile_motes = @sapient_sr_base }
}

subplan = {
	set_name = "Strategic Volatile Motes (Refinery)"
	potential = {
		has_technology = tech_volatile_motes
		NOT = { any_owned_planet = { has_volatile_motes_deposit = yes } }
		NOT = { has_monthly_income = { resource = volatile_motes value >= @sapient_sr_refinery_base } }
	}
	income = { volatile_motes = @sapient_sr_refinery_base }
}
