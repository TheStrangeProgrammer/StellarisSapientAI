# Consumer goods — base floor only. sapient_uses_consumer_goods fixes vanilla's
# expense-based country_uses_consumer_goods (same leaks: machines during the
# uninitialised window, transient conquered organics). Obsessional-directive
# machines — the one gestalt CG user — stay in.
subplan = {
	set_name = "Base Consumer Goods"
	potential = { sapient_uses_consumer_goods = yes }
	income = { consumer_goods = @sapient_consumer_goods_base }
}

# Obsessional Directive ("paperclip AI"): it MUST produce consumer goods, and
# every 15k CG converts into a large lump of unity/research and drains 15k from the
# stockpile. CG is therefore its CORE OUTPUT — more is always better, 1:1 into
# unity/research with no diminishing return, exactly like research is for a tech
# empire. So scale it UNBOUNDED (no stockpile cap): the stockpile never rests full
# (it's dumped every 15k, so a cap would rarely trip), and a cap would kneecap the
# empire's whole economy in the endgame. Bounded only by the treadmill's own
# "fulfilled" self-limit — i.e. by what the empire can actually produce each pass.
subplan = {
	scaling = yes
	set_name = "Scaling Consumer Goods (Obsessional)"
	potential = {
		has_valid_civic = civic_machine_obsessional_directive
	}
	income = { consumer_goods = @sapient_consumer_goods_step }
}

# --- Basic buffer treadmill (capped) --------------------------------------
# Climbs one district at a time up to @sapient_consumer_goods_scaling_cap, so the empire
# carries a real surplus cushion instead of budgeting to a knife's edge. Stops at
# the cap -- late game the buffer is noise, so it stops costing districts.
subplan = {
	scaling = yes
	set_name = "Scaling Consumer Goods (Buffer)"
	potential = {
		sapient_uses_consumer_goods = yes
		NOT = { has_monthly_income = { resource = consumer_goods value >= @sapient_consumer_goods_scaling_cap } }
	}
	income = { consumer_goods = @sapient_consumer_goods_step }
}
