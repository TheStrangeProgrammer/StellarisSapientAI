			factor = 1.5

			factor = 2

		}

			}
		}
		# Bioship Starhold costs 500 alloys - base 500 is sufficient, no add needed
		modifier = {
			# Allow saving enough for a Starfortress (2500 alloys)
			add = 2000
			country_uses_bio_ships = no
			has_technology = tech_starbase_4
			any_owned_starbase = {
				has_starbase_size = starbase_starhold
				can_be_upgraded = yes
			}
		}
		modifier = {
			# Allow saving enough for a Bioship Starfortress (1250 alloys)
			add = 750
			country_uses_bio_ships = yes
			has_technology = tech_starbase_4
			any_owned_starbase = {
				has_starbase_size = starbase_starhold
				can_be_upgraded = yes
			}
		}
		modifier = {
			# Allow saving enough for a Citadel (5000 alloys)
			add = 4500
			country_uses_bio_ships = no
			has_technology = tech_starbase_5
			any_owned_starbase = {
				has_starbase_size = starbase_starfortress
				can_be_upgraded = yes
			}
		}
		modifier = {
			# Allow saving enough for a Bioship Citadel (2500 alloys)
			add = 2000
			country_uses_bio_ships = yes
			has_technology = tech_starbase_5
			any_owned_starbase = {

alloys_expenditure_starbases_expand = {
	resource = alloys
	type = expenditure
	category = starbases

	potential = {
		sapient_expansion_wanted = yes
		# SapientAI: was `highest_threat < 50` — highest_threat only rises AFTER
		# aggression happens (see 20_sapient_world_triggers.txt), so it never
		# paused expansion when it mattered. Use our intent+power ladder instead:
		# pause frontier outposts once someone credibly intends to attack (or we
		# are prepping our own offensive and want alloys on the fleet).
		sapient_threat_alarmed = no
		NOT = {
			is_country_type = fallen_empire
			is_country_type = awakened_fallen_empire
		}
		# SapientAI: vanilla disabled outpost spending whenever ANY colonization
		# (or wilderness terraform) plan was queued. For empires whose colony
		# ships cost this same resource that deadlocks: the colony target can
		# sit outside borders, needing the outpost this entry is forbidden from
		# funding (observed: Rogue Servitor hoarding alloys, zero expansion).
		# Plans now only suppress expansion while the empire cannot afford both.
		OR = {
			NOT = { sapient_colonization_underway = yes }
			has_resource = { type = alloys amount > 800 }
		}
		has_resource = { type = influence amount > 75 }
	}

	weight = {
		weight = 0.2

		# Biological Starbases use less alloys
		modifier = {
			factor = 0.5
			country_uses_bio_ships = yes
		}
	}

	desired_min = {
		# SapientAI: raised from vanilla 150 — reserve for a QUEUE of outposts
		# (~90-130 alloys each), not one, while the expansion phase runs.
		base = 250
	}
}

alloys_expenditure_ships = {
	resource = alloys
	type = expenditure
	category = ships

	potential = {
		always = yes
	}

	weight = {
		weight = 0.6

		# SapientAI expansion-phase gate. TESTED twice:
		#   factor 0.1 failed — 10 corvettes built anyway. Mechanism: when
		#   sibling categories sit at desired_max (starbases 500), OVERFLOW
		#   redistributes by weight among uncapped categories; ships absorbed
		#   it all despite a tiny share (observed trigger: 610 = 500 + one
		#   corvette). Zero weight receives zero redistribution — surplus
		#   pools in the bank instead, which climbs through the expansion
		#   band and releases at 800: the stockpile IS the phase signal.
		#   Ships exist only when alloys accumulate faster than expansion
		#   can spend them. War or threat releases immediately.
		modifier = {
			factor = 0
			sapient_threat_wary = no
			has_resource = { type = alloys amount < 800 }
			# frontier scan last: lazy AND means rich empires exit before it
			sapient_expansion_wanted = yes
		}

		# Graduated war escalation (replaces the flat is_at_war x3). Mutually-
		# exclusive threat bands ramp ship spending up to the at-war peak; the
		# lower rungs are suppressed while recently_lost_war so the rebuild boost
		# (peak) never double-multiplies with them.
		modifier = {
			factor = @sapient_ship_war_wary
			sapient_threat_wary = yes
			sapient_threat_alarmed = no
			recently_lost_war = no
		}
		modifier = {
			factor = @sapient_ship_war_alarmed
			sapient_threat_alarmed = yes
			sapient_threat_mobilizing = no
			recently_lost_war = no
		}
		modifier = {
			factor = @sapient_ship_war_mobilizing
			sapient_threat_mobilizing = yes
			sapient_threat_war = no
			recently_lost_war = no
		}
		modifier = {
			factor = @sapient_ship_war_peak
			OR = {
				sapient_threat_war = yes
				recently_lost_war = yes
			}
		}

		# spend less alloys on ships if over capacity
		modifier = {
			factor = 0.75
			used_naval_capacity_percent >= 1.0
		}

		# Biological Ships use less alloys
		modifier = {
			factor = 0.33
			country_uses_bio_ships = yes
		}

		modifier = {
			factor = 5
			used_naval_capacity_percent < 1.5
			OR = {
				any_neighbor_country = {
					has_ascension_perk = ap_become_the_crisis
				}
				any_country = {
					is_crisis_faction = yes
				}
			}
		}
	}

	desired_min = {
		base = 0

		modifier = {
			add = @ai_ship_budget_desired_min_per_year
			country_uses_bio_ships = no
		}

		modifier = {
			factor = trigger:years_passed
		}

		# SapientAI: desired_min is a guideline OVERRIDE of the fraction
		# system — it would keep dripping ship allocation past a zero
		# weight. Zeroed under the same expansion-phase gates.
		modifier = {
			factor = 0
			sapient_threat_wary = no
			has_resource = { type = alloys amount < 800 }
			# frontier scan last: lazy AND means rich empires exit before it
			sapient_expansion_wanted = yes
		}
	}
}
