# SapientAI — industrial input scaling template
#
# One template, instantiated per archetype from ai_base.txt:
#   $NAME$     display name of the subplan (unique per instantiation)
#   $GATE$     archetype scripted trigger (10_sapient_archetypes.txt)
#   $RESOURCE$ the input resource this archetype's industry consumes
#   $STEP$     income added per scaling iteration
#   $LIMIT$    monthly-income ceiling where scaling stops
#
# Shape mirrors vanilla's "Base Minerals Scaling" exactly (nomad exclusion,
# income ceiling, stockpile brake) so behavior for the mineral archetype is
# identical to vanilla by construction.

subplan = {
	scaling = yes
	set_name = "$NAME$"
	potential = {
		$GATE$ = yes
		NOR = {
			is_nomadic = yes
			has_monthly_income = { resource = $RESOURCE$ value >= $LIMIT$ }
			resource_stockpile_compare = {
				resource = $RESOURCE$
				value > @economic_plan_stockpile_threshold_high
			}
		}
	}

	income = {
		$RESOURCE$ = $STEP$
	}
}
