# Alloys — bioships still buy full-price EQUIPMENT (weapons/armor/shields) in
# alloys; only the hull is offloaded to food. So bioships get a lower base (10 vs
# 15) and lower caps, but the SAME scaling step as normal shipbuilders. Nemesis
# excluded (crisis ships aren't alloy-built).
#
# NOTE: subplan ai_weight is inert (vanilla 00_example.txt), so the old
# per-subplan weight/deficit/threat modifiers were dead code and are removed.
# Deficit priority is automatic: a base subplan is an always-added target, and
# an unfulfilled target is what drives the AI to close the gap. War pressure is
# re-expressed below as real (bounded) war subplans, not inert weights.

# --- Base floors ----------------------------------------------------------
subplan = {
	set_name = "Base Alloys"
	potential = { sapient_is_bioship_empire = no }
	income = { alloys = @sapient_alloys_base }
}

# Bioship alloy base: equipment only (hull is food) -> lower floor than normal.
subplan = {
	set_name = "Base Alloys (Bioship)"
	potential = { sapient_is_bioship_empire = yes }
	income = { alloys = @sapient_alloys_base_bioship }
}

# --- Peacetime treadmill --------------------------------------------------
subplan = {
	scaling = yes
	set_name = "Scaling Alloys (Normal)"
	potential = {
		sapient_scales_alloys_normal = yes
		NOT = { sapient_alloys_over_cap = yes }
	}
	income = { alloys = @sapient_alloys_step }
}

subplan = {
	scaling = yes
	set_name = "Scaling Alloys (Bioship)"
	potential = {
		sapient_scales_alloys_bioship = yes
		NOT = { sapient_alloys_over_cap_bioship = yes }
	}
	income = { alloys = @sapient_alloys_step_bioship }
}

# --- War escalation (normal shipbuilders) ---------------------------------
# Real subplans: add a bounded amount on top of the base while the rung holds.
# Cumulative (rungs nest). All @...=0 until tuned (see 10_sapient_variables).
subplan = {
	set_name = "War Alloys (Wary)"
	potential = { sapient_scales_alloys_normal = yes  sapient_threat_wary = yes }
	income = { alloys = @sapient_alloys_war_wary }
}
