# SapientAI — production chain with strategic-resource coupling
# Same as production_chain, plus a third coupled resource: the SR that this
# chain's tier buildings cost and upkeep (foundries: volatile motes; factories:
# rare crystals — 04_manufacturing_buildings.txt). SR demand thereby grows with
# industrial scale instead of vanilla's flat optional 1/month.
# Extra params: $SR$, $SR_STEP$

subplan = {
	scaling = yes
	set_name = "$NAME$"
	potential = {
		$GATE$ = yes
		country_uses_bio_ships = no
		NOT = { has_ascension_perk = ap_become_the_crisis }
		meets_basic_resource_thresholds = yes
		has_monthly_income = { resource = $OUTPUT$ value < $OUT_LIMIT$ }
	}

	income = {
		$OUTPUT$ = $OUT_STEP$
		$INPUT$ = $IN_STEP$
		$SR$ = $SR_STEP$
	}
}
