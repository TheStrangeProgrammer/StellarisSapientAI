# Energy — base floor for all; pure-consumption resource, so the surplus target
# already covers upkeep and rising upkeep raises the bar automatically. No
# peacetime scaling (nothing dumps energy into growth). Wartime fleet upkeep for
# standard/machine ships IS energy, so war tiers add a bounded top-up (bioships
# burn food instead — see plan_food). Deficit priority is automatic (unfulfilled
# target drives the fix); the old inert deficit/threat weights are removed.
subplan = {
	set_name = "Base Energy"
	# is_nomadic=no matches vanilla ("Base Minerals & Energy" is nomad-gated).
	# Without it, nomads would get this floor PLUS the nomad energy subplan in
	# 01_base.txt on top (double-counted).
	potential = { is_nomadic = no }
	income = { energy = @sapient_energy_base }
}

# --- Basic buffer treadmill (capped) --------------------------------------
# Climbs one district at a time up to the tech-tiered ceiling (sapient_energy_buffer_over_cap), so the empire
# carries a real surplus cushion instead of budgeting to a knife's edge. Stops at
# the cap -- late game the buffer is noise, so it stops costing districts.
subplan = {
	scaling = yes
	set_name = "Scaling Energy (Buffer)"
	potential = {
		is_nomadic = no
		# Bioships pay FLEET upkeep in food, not energy, so a big energy cushion is
		# dead weight for them — they get the food buffer instead (plan_food). They
		# keep the flat Base Energy floor for building/pop upkeep.
		sapient_is_bioship_empire = no
		# Tech-tiered stop (was a flat 200 cap, which ate the early build
		# order): 75/125/225 income ceiling by power-plant tech rung.
		NOT = { sapient_energy_buffer_over_cap = yes }
	}
	income = { energy = @sapient_energy_step }
}

# --- Purge windfall guard --------------------------------------------------
# The buffer above measures net income, so a purge windfall SATISFIES it for
# free and the treadmill idles while the ghost income lasts (see the guard
# block in 10_sapient_variables for the full failure chain). This flat lane
# arms only while a purge is running AND income is windfall-inflated, and
# demands enough to push the target above the windfall — pure build pressure
# that turns ghost income into districts, self-cancelling when the purge ends.
# NOT a cap bypass: the tech-tiered treadmill stop above still stands.
# Keyed to MACHINE purgers only — purge_matrix is the only purge job that
# pays energy. An organic purifier's purge windfall is food (or minerals for
# lithoid prey), so arming this guard for them would overbuild generators
# during every purge for nothing. Mirror lanes: plan_food, plan_minerals.
subplan = {
	set_name = "Purge Windfall Guard (Energy)"
	potential = {
		sapient_purge_windfall_energy = yes
	}
	income = { energy = @sapient_purge_windfall_guard }
}
