# INDUSTRY input (foundry/artisan): minerals | food (catalytic).
# RESEARCH input (calculator_* jobs, all three disciplines identical):
#   individualist CG 1.5/3 | machine energy 3/3 | hive minerals 4/3
#   (lithoid-INDEPENDENT for research).
# UNITY input (bureaucrat / coordinator):
#   individualist CG 2/3 | machine energy 4/3
#   hive-bio food 2 + energy 2 /3 | hive lithoid-or-infernal minerals 2 + energy 2 /3
#   (unity DOES split on is_lithoid_or_infernal_empire — vanilla trigger).

sapient_industry_input_is_minerals = {
	is_catalytic_empire = no
}
sapient_industry_input_is_food = {
	is_catalytic_empire = yes
}

sapient_research_input_is_cg = {

	is_gestalt = no
}
sapient_research_input_is_energy = {

}

# --- CG chain gates (audit fix) -------------------------------------------
# Vanilla CG subplans gated country_uses_consumer_goods and excluded the
# machine-obsessional civic (it has its own dedicated scaling, kept in
# plan_cg). Our chains initially dropped both conditions — restored here.

sapient_cg_chain_minerals = {
	sapient_industry_input_is_minerals = yes
	country_uses_consumer_goods = yes
	NOT = { has_valid_civic = civic_machine_obsessional_directive }
}
sapient_cg_chain_food = {
	sapient_industry_input_is_food = yes
	country_uses_consumer_goods = yes
	NOT = { has_valid_civic = civic_machine_obsessional_directive }
}
