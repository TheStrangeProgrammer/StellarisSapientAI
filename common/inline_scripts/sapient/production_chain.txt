# SapientAI — production chain template
#
# A chain couples an output and its input in ONE scaling subplan, so both
# targets rise in the same tick and get built in parallel. This is the fix for
# build-time serialization (480d building lag compounding into permanent
# deficit-chasing), and for vanilla's under-provisioned ceilings: vanilla caps
# alloys at 250 AND minerals at 250, but 250 alloys requires ~500+ minerals of
# input at the 2:1 job ratio. Coupling makes input demand derived, not capped.
#
# Parameters (inline substitution is textual — no arithmetic — so input steps
# are precomputed at the instantiation site; see manifest in ai/ai_base.txt):
#   $NAME$       unique subplan name
#   $GATE$       archetype scripted trigger, or 'always'
#   $OUTPUT$     produced resource
#   $INPUT$      consumed resource for this archetype
#   $OUT_STEP$   output income added per scaling iteration
#   $IN_STEP$    input income added per iteration
#                = OUT_STEP x job ratio x 1.25 over-provision buffer
#                  (the buffer absorbs build latency: input capacity must lead
#                   output capacity, since the system cannot schedule)
#   $OUT_LIMIT$  output monthly-income ceiling where the chain stops scaling
#
# Standing conditions mirror vanilla's "Base Alloy Scaling": chains describe
# standard planetary industry, so bioship economies and Nemesis crisis empires
# are excluded, and chains only climb when basic resources are solvent.

subplan = {
	scaling = yes
	set_name = "$NAME$"
	potential = {
		$GATE$ = yes
		country_uses_bio_ships = no
		NOT = { has_ascension_perk = ap_become_the_crisis }
		meets_basic_resource_thresholds = yes
		has_monthly_income = { resource = $OUTPUT$ value < $OUT_LIMIT$ }
	}

	income = {
		$OUTPUT$ = $OUT_STEP$
		$INPUT$ = $IN_STEP$
	}
}
