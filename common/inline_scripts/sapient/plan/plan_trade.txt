# Trade — the unconditional base floor is shared by ALL empires (trade is
# consumed everywhere; gestalts produce it too, they're just not optimized for
# it), so it lives in the top-level income block of 01_base.txt (vanilla puts
# trade there too). Only the megacorp scaling lane remains here — megacorps build
# their economy on trade as a growth engine.
subplan = {
	scaling = yes
	set_name = "Scaling Trade (Megacorp)"
	potential = {
		is_megacorp = yes
		NOT = { resource_stockpile_compare = { resource = trade value >= @sapient_trade_stock_cap } }
	}
	income = { trade = @sapient_trade_step }
}
