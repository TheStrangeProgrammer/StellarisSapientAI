# SapientAI — trade budget overrides (same-name key replace).
# Colony-ship entries: activation swapped from raw ai_colonize_plans to
# sapient_colonization_underway (L0 trigger: plan exists AND reachable
# in-border target). Vanilla reserved colony funds for targets OUTSIDE
# borders — starving expansion of exactly the money needed to reach them.
# Now: unreachable plan reserves nothing -> expansion claims the system ->
# target enters borders -> trigger flips -> colony reserve activates.

trade_expenditure_colonies_sponsored = {
	resource = trade
	type = expenditure
	category = colonies

	potential = {
		is_nomadic = no
		is_guided_sapience_empire = no
		OR = {
			has_civic = civic_corporate_dominion
			has_civic = civic_private_prospectors
		}
		sapient_colonization_underway = yes
		has_resource = { type = trade amount > 400 }
	}

	weight = {
		weight = 1.5
	}

	desired_min = {
		base = @colony_cost_high # Sponsored colonizer costs 500 trade
	}

	desired_max = {
		base = @colony_cost_high_max
	}
}

# --- Market rescue rework --------------------------------------------------
# Vanilla's two market pools create a DEAD ZONE: the general pool needs 2500+
# banked trade, the deficit pool needs income < -1 on the starved resource.
# An AI at +4 minerals with 1.2k trade satisfies neither — it starves under a
# flood of mineral-hungry plans while the trade sits idle. Observed in-game.
#
# Rescue taxonomy (design ruling): only the BUILD-METHOD resource rescues at
# "income too low to function" (< 10 — +4 can't fund one district a year);
# consumables (energy/food/CG) only rescue on actual deficit + low bank, as
# vanilla — buying upkeep you aren't short of is wasted trade. Build method is
# minerals for everyone except nomads, whose building resource is ALLOYS, so
# they get a mirrored alloys branch; the universal alloys deficit branch stays
# vanilla for non-nomads. General pool gate 2500 -> 1000 (pure hoarding).
# desired_min charges a spendable lump instead of a weight-share trickle.

trade_expenditure_trade = {
	resource = trade
	type = expenditure
	category = trade

	potential = {
		has_resource = {
			type = trade
			amount > 1000	# vanilla 2500
		}
	}

	weight = {
		weight = 0.5
	}
}

trade_expenditure_trade_deficit = {
	resource = trade
	type = expenditure
	category = trade

	potential = {
		has_resource = {
			type = trade
			amount > 100
		}
		OR = {
			# build method, standard empires: minerals
			AND = {
				is_nomadic = no
				has_monthly_income = {
					resource = minerals
					value < 10	# vanilla < -1: the +4 deadlock case
				}
				has_resource = {
					type = minerals
					amount < 1000
				}
			}
			# build method, nomads: alloys
			AND = {
				is_nomadic = yes
				has_monthly_income = {
					resource = alloys
					value < 10
				}
				has_resource = {
					type = alloys
					amount < 1000
				}
			}
			# consumables below: vanilla deficit-only rescue
			AND = {
				has_monthly_income = {
					resource = energy
					value < -1
				}
				has_resource = {
					type = energy
					amount < 1000
				}
			}
			AND = {
				has_monthly_income = {
					resource = food
					value < -1
				}
				has_resource = {
					type = food
					amount < 500
				}
			}
			AND = {
				has_monthly_income = {
					resource = consumer_goods
					value < -1
				}
				has_resource = {
					type = consumer_goods
					amount < 500
				}
			}
			AND = {
				has_monthly_income = {
					resource = alloys
					value < -1
				}
				has_resource = {
					type = alloys
					amount < 500
				}
			}
		}
	}

	weight = {
		weight = 0.7
		# starving on the BUILD resource is an emergency, not a line item
		modifier = {
			factor = 2
			OR = {
				AND = {
					is_nomadic = no
					has_monthly_income = { resource = minerals value < 5 }
				}
				AND = {
					is_nomadic = yes
					has_monthly_income = { resource = alloys value < 5 }
				}
			}
		}
	}

	desired_min = {
		base = 250
	}

	desired_max = {
		base = 1000
	}
}
