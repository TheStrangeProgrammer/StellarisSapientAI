# sapient_unity_arc already folds in both the perk cut-off (num_ascension_perks <
# cap) AND the rogue-servitor exclusion (bio-trophies generate unity passively;
# servitors can't build it). Unity is spent, never banked, so the treadmill
# self-limits on surplus rather than a stockpile cap.

	scaling = yes
	set_name = "Scaling Unity (Unity Arc)"
	potential = { sapient_unity_arc = yes }
	income = { unity = @sapient_unity_step_unity_arc }
