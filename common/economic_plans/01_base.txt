# SapientAI — basic economy plan (full replacement of vanilla 01_base.txt)
# The plan is a manifest; all demand lives in per-resource inlines, each of
# which defines that resource's production chains and conditional variants.

basic_economy_plan = {
	ai_weight = { weight = 1000 }
	# Top-level income = UNCONDITIONAL floors only, mirroring vanilla ai_base
	# (which puts alloys/research/unity/trade here). In THIS mod only alloys stays
	# out — it's bioship-split (10/15) and can't be a single value, so it lives as
	# subplans in plan_alloys. Unity, research, and trade are unconditional floors
	# shared by every empire (gestalts produce trade too, just aren't optimized for
	# it) and live here. Energy, minerals, food, cg, and SR are gated and carried
	# by their own inline script's base subplan. (Zero entries removed — a 0 target
	# is inert; vanilla lists only what it actually targets.)
	income = {
		unity = @sapient_unity_base
		trade = @sapient_trade_base
		physics_research = @sapient_research_base
		society_research = @sapient_research_base
		engineering_research = @sapient_research_base
	}

subplan = {
		potential = {
			is_nomadic = yes
		}
		set_name = "Base Minerals & Energy Nomads"
		income = {
			minerals = @economic_plan_mineral_target_nomads
			energy = @economic_plan_energy_target_nomads
		}
	}

	inline_script = "sapient/plan/plan_minerals"
	inline_script = "sapient/plan/plan_energy"
	inline_script = "sapient/plan/plan_food"
	inline_script = "sapient/plan/plan_alloys"
	inline_script = "sapient/plan/plan_cg"
	inline_script = "sapient/plan/plan_unity"
	inline_script = "sapient/plan/plan_research"
	inline_script = "sapient/plan/plan_trade"
	inline_script = "sapient/plan/plan_pops"
	inline_script = "sapient/plan/plan_naval"
	inline_script = "sapient/plan/plan_strategic"
}
