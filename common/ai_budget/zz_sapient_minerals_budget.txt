minerals_expenditure_colonies_lithoid_expand = {
	resource = minerals
	type = expenditure
	category = colonies

	potential = {
		is_nomadic = no
		is_robot_empire = no
		is_lithoid_empire = yes
		# SapientAI: vanilla's ONLY colony entry with no colonization gate at
		# all — lithoids carried a permanent colony-ship mineral reservation
		# from 400 banked onward, plans or none, competing with construction.
		# Same trigger as every sibling entry: reserve only for reachable
		# targets; expansion->colonization handoff applies.
		sapient_colonization_underway = yes
		NOT = {
			has_origin = origin_synthetic_fertility
		}
		has_resource = { type = minerals amount > 400 }
	}

	weight = {
		weight = 0.8
	}

	desired_min = {
		base = 0
		modifier = {
			add = @colony_cost_base
			is_hive_empire = no
		}
		modifier = {
			add = @colony_cost_high
			OR = {
				is_hive_empire = yes
				has_origin = origin_lithoid
			}
		}
	}

	desired_max = {
		base = 0
		modifier = {
			add = @colony_cost_base_max
			is_hive_empire = no
		}
		modifier = {
			add = @colony_cost_high_max
			OR = {
				is_hive_empire = yes
				has_origin = origin_lithoid
			}
		}
	}
}

minerals_expenditure_ships = {
	resource = minerals
	type = expenditure
	category = ships

	potential = {
		OR = {
			uses_fauna_ship_sizes = yes
			AND = {
				has_ascension_perk = ap_become_the_crisis
				has_crisis_level = crisis_level_2
			}
		}
	}

	weight = {
		weight = 0.4

		# SapientAI expansion-phase gate (mirrors alloys ships; see that file
		# for the tested overflow-redistribution rationale).
		modifier = {
			factor = 0
			sapient_threat_wary = no
			has_resource = { type = minerals amount < 800 }
			# frontier scan last: lazy AND means rich empires exit before it
			sapient_expansion_wanted = yes
		}

		# Graduated war escalation (replaces the flat is_at_war x2). Crisis/Nemesis
		# minerals ships run a lower curve than standard ships and keep their own
		# peak of 2 (no recently_lost_war band — the original had none). Mutually
		# exclusive bands; see alloys budget for the tier mapping.
		modifier = {
			factor = 1.25
			sapient_threat_wary = yes
			sapient_threat_alarmed = no
		}
		modifier = {
			factor = 1.5
			sapient_threat_alarmed = yes
			sapient_threat_mobilizing = no
		}
		modifier = {
			factor = 1.75
			sapient_threat_mobilizing = yes
			sapient_threat_war = no
		}
		modifier = {
			factor = 2
			sapient_threat_war = yes
		}

		# spend less minerals on ships if over capacity
		modifier = {
			factor = 0.5
			used_naval_capacity_percent >= 1.0
		}
	}
}
